Controllers

Routes can be grouped in a Controller.

Example:

{% highlight java %}public class UserController extends Controller { { get("/users", this::listUsers); post("/users", this::addUser); put("/users/{id}", this::updateUser); patch("/users/{id}", this::updateUser); delete("/users/{id}", this::deleteUser); } public CompletableFuture listUsers(Request request) { ... } public CompletableFuture addUser(Request request) { ... } public CompletableFuture updateUser(Request request) { ... } private CompletableFuture deleteUser(Request request) { ... } public static void main(String[] args) { new HttpServer() .register(new UserController()) .start(); } }{% endhighlight %}
