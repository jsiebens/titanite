Titanite

A small web framework for Java 8, inspired by others like Sinatra and Finatra, running on top of Netty

{% highlight java %}import static org.nosceon.titanite.Titanite.*; import static org.nosceon.titanite.Method.*; public class HelloWorld { public static void main(String[] args) { httpServer() .register(GET, "/hello/{name}", r -> { String name = r.pathParams.get("name"); return ok().text("hello " + name).toFuture(); }) .start(8080); } }{% endhighlight %}
